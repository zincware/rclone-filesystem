2. Rewrite `ls()` method:
   ```python
   def ls(self, path, detail=False, **kwargs):
       path = self._strip_protocol(path).rstrip("/")
       refresh = kwargs.pop("refresh", False)

       # Check cache first (unless refresh requested)
       if not refresh and path in self.dircache:
           entries = self.dircache[path]
       else:
           rclone_path = self._make_rclone_path(path)
           result = rclone.ls(rclone_path, max_depth=1)
           entries = [
               {
                   "name": (path + "/" + x["Path"]).lstrip("/"),
                   "size": x["Size"],
                   "type": "directory" if x["IsDir"] else "file",
                   "ModTime": x.get("ModTime"),
                   "MimeType": x.get("MimeType"),
               }
               for x in result
           ]
           # Populate DirCache
           self.dircache[path] = entries

           # FNFE heuristic: empty result -> check parent
           if not entries:
               self._raise_if_not_found(path)

       if detail:
           return entries
       return sorted([e["name"] for e in entries])
   ```
   - Remove the old docstring about "will not raise FileNotFoundError" limitation
   - Always pass `max_depth=1` to `rclone.ls()` to prevent recursive listing (Pitfall 7)
   - Always `.rstrip("/")` path before using as DirCache key (Pitfall 2)

3. Implement `info()` method:
   ```python
   def info(self, path, **kwargs):
       path = self._strip_protocol(path).rstrip("/")

       # Check if parent is cached -- entry might be in parent listing
       parent = self._parent(path)
       if parent in self.dircache:
           for entry in self.dircache[parent]:
               if entry["name"].rstrip("/") == path:
                   return entry

       # Not in cache -- call rclone for this path
       rclone_path = self._make_rclone_path(path)
       try:
           result = rclone.ls(rclone_path, max_depth=1)
       except RcloneException as e:
           raise FileNotFoundError(f"No such file or directory: '{path}'") from e

       if not result:
           raise FileNotFoundError(f"No such file or directory: '{path}'")

       # If result has entries, path is a directory (rclone lists contents)
       # Construct appropriate entry
       # For a file: rclone.ls on the parent would have returned it
       # For a directory: rclone.ls on the path returns its contents
       # We need to determine if path is a file or directory
       # Strategy: if we get entries back, path is a directory
       # (rclone.ls lists *contents* of the path)
       entry = {
           "name": path,
           "size": 0,
           "type": "directory",
       }

       # But wait -- if path is a file, rclone.ls on file's parent with max_depth=1
       # would list it. We need to check parent to get file info.
       # Alternative: ls the parent, find our entry
       parent_result = rclone.ls(self._make_rclone_path(parent), max_depth=1)
       basename = path.rsplit("/", 1)[-1] if "/" in path else path
       for x in parent_result:
           if x["Path"] == basename:
               entry = {
                   "name": path,
                   "size": x["Size"],
                   "type": "directory" if x["IsDir"] else "file",
                   "ModTime": x.get("ModTime"),
                   "MimeType": x.get("MimeType"),
               }
               break

       # Cache the parent listing while we have it
       parent_entries = [
           {
               "name": (parent + "/" + x["Path"]).lstrip("/"),
               "size": x["Size"],
               "type": "directory" if x["IsDir"] else "file",
               "ModTime": x.get("ModTime"),
               "MimeType": x.get("MimeType"),
           }
           for x in parent_result
       ]
       self.dircache[parent] = parent_entries

       return entry
   ```

   NOTE: The exact info() implementation is at Claude's discretion per CONTEXT.md. The key contracts are:
   - Check DirCache first (parent entry lookup)
   - If not cached, call rclone and determine file/directory status
   - Populate DirCache with fetched data
   - Raise FileNotFoundError for nonexistent paths
   - Return dict with at minimum: name, size, type

4. Remove the existing `ls()` docstring limitation note about not raising FileNotFoundError.
